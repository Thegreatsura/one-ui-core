The `EmojiPickerDropdown` component combines the `EmojiPicker` with a `DropdownWrapper` to provide a complete dropdown emoji selection experience. Simply pass a trigger element and handle emoji selection.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <BasicEmojiPickerDropdown />
  }
  codes={[
    {
      code:
`const [selectedEmoji, setSelectedEmoji] = useState("");

<Row gap="16" vertical="center">
  <EmojiPickerDropdown
    trigger={<Button label="Select Emoji" />}
    onSelect={(emoji) => setSelectedEmoji(emoji)}
  />
  {selectedEmoji && (
    <Text variant="heading-default-xl">
      Selected: {selectedEmoji}
    </Text>
  )}
</Row>`,
      language: "tsx",
      label: "EmojiPickerDropdown"
    }
  ]}
/>

## Controlled state

Control the open/close state externally using `isOpen` and `onOpenChange`.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <ControlledEmojiPickerDropdown />
  }
  codes={[
    {
      code:
`const [isOpen, setIsOpen] = useState(false);
const [selectedEmoji, setSelectedEmoji] = useState("");

<Row gap="16" vertical="center">
  <EmojiPickerDropdown
    trigger={<Button label="Controlled Dropdown" />}
    onSelect={(emoji) => setSelectedEmoji(emoji)}
    isOpen={isOpen}
    onOpenChange={setIsOpen}
  />
  <Button 
    variant="secondary" 
    label={isOpen ? "Close" : "Open"} 
    onClick={() => setIsOpen(!isOpen)}
  />
  {selectedEmoji && (
    <Text variant="heading-default-xl">
      {selectedEmoji}
    </Text>
  )}
</Row>`,
      language: "tsx",
      label: "Controlled state"
    }
  ]}
/>
