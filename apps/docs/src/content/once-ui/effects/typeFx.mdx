TypeFx creates a realistic typewriter animation that types out text character-by-character. When multiple words are provided, it automatically types, holds, deletes, and cycles through them.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <Row gap="8" vertical="center">
      <TypeFx
        words={["amazing products", "stunning websites", "modern apps"]}
        speed={80}
        hold={2000}
        trigger="instant"
      >Build{" "}</TypeFx>
    </Row>
  }
  codes={[
    {
      code:
`<Row gap="8" vertical="center">
  <Text variant="heading-default-l">

    speed={80}
    hold={2000}
    trigger="instant"
  />
</Row>`,

    }
  ]}
/>

## Custom trigger

Use the `custom` trigger to programmatically control when the typing animation starts.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <Column gap="16">
      <TypeFxCustomExample />
    </Column>
  }
  codes={[
    {
      code:
`function TypeFxCustomExample() {
  const triggerFnRef = useRef<(() => void) | null>(null);

  return (
    <>
      <TypeFx
        words={["Click the button", "to start typing", "this animation"]}
        speed={60}
        hold={1500}
        trigger="custom"
        onTrigger={(fn) => {
          triggerFnRef.current = fn;
        }}
      />
      <Button onClick={() => triggerFnRef.current?.()}>
        Start typing
      </Button>
    </>
  );
}`,
      language: "tsx",
      label: "Custom trigger"
    }
  ]}
/>
