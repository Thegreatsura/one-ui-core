The `MatrixFx` component creates an animated dot matrix reveal effect using the native Canvas 2D API. It supports instant reveal, progressive reveal on mount, or hover-triggered animations with smooth transitions. Perfect for hero sections, backgrounds, or interactive hover cards.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  previewPadding="0"
  preview={
    <MatrixFx
      height={24}
      colors={["accent-solid-strong", "accent-solid-weak"]}
      trigger="mount"
      flicker
    />
  }
  codes={[
    {
      code:
`<MatrixFx
    height={24}
    colors={["brand-solid-strong", "accent-solid-strong"]}
    trigger="mount"
    flicker
/>`,
      language: "tsx",
      label: "MatrixFx"
    }
  ]}
/>
